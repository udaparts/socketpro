[propget, id(18), helpstring("property ConnTimeout")] HRESULT ConnTimeout([out, retval] long *pVal);

[propput, id(18), helpstring("property ConnTimeout")] HRESULT ConnTimeout([in] long newVal);

 

Remarks

 

This property determines the timeout for building a socket connection to a remote host. By default, it is 30,000 ms.
