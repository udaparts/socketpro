[id(99), helpstring("method GetInfo")] HRESULT GetInfo([in]long lIndex, [out]long *plGroup, [out]BSTR *pbstrUID, [out]long *plSvsID, [out]long *plPort, [out, retval]BSTR *pbstrIPAddress);

 

Parameters

 

lIndex

            [in]Zero-based index

 

plGroup

            [out]A pointer to an integer for receiving a chat group id

 

pbstrUID

            [out]A pointer to a BSTR for receiving a user id of a client

 

plSvsID

            [out]A pointer to an integer for receiving the current service id of the client

 

plPort

            [out]A pointer to an integer for receiving the port number of the client

 

pbstrIPAddress

            [out]A pointer to a BSTR for receiving the IP address of the client in the format “111.222.212.121”

 

 

Remarks

 

            Calling the function queries the information of a client that sends out of a chat requests, Enter, Exit, Speak, SpeakTo, GetAllListeners, GetAllClients, SpeakEx and SpeakToEx.

 

            lIndedx should be always 0 for the requests, Enter, Exit, Speak, SpeakTo, SpeakEx and SpeakToEx.
