enum tagSSLEvent

{

            ssleHandshakeStarted = 0x10, //Handshake started

            ssleHandshakeDone = 0x20, //Handshake completed

            ssleHandshakeLoop = 0x1001, //Handshake looping

            ssleHandshakeExit = 0x1002, //Handshake exiting

            ssleReadAlert = 0x4004, //Reading data

            ssleWriteAlert = 0x4008, //Writing data

            ssleDoSSLConnect = 0x811F, //Begin SSL/TLSv1 connecting

};

 

Remarks

 

            See the comments for each of enum members. The enum is used to indicate SSL/TLSv1 handshake states by use of either openSSL or MS SSPI.
