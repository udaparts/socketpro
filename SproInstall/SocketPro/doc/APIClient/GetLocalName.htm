[id(11), helpstring("method GetLocalName")] HRESULT GetLocalName([out, retval] BSTR *pbstrLocalName);

 

Parameter

 

pbstrLocalName

            [out]A pointer to a BSTR for receiving local machine name.

 

Remarks

 

            This function returns the standard host name for the local machine. Refer to socket function gethostname.
