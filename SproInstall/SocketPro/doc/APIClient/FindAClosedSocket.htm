id(12), helpstring("method FindAClosedSocket")] HRESULT FindAClosedSocket([out, retval]IUSocket **ppIUSocket);

 

Parameter

 

ppIUSocket

            [out]A pointer to an interface for receiving the interface IUSocket

 

Remarks

 

Calling this function finds a USocket object that is not connected to a remote SocketPro server.

 

Note that the reference of the pointed USocket object will be increased by 1 if *ppIUSocket is not NULL.
