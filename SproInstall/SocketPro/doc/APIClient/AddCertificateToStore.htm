[id(218), helpstring("method AddCertificateToStore")] HRESULT AddCertificateToStore([in, defaultvalue(0)]VARIANT_BOOL bRoot);

 

Parameter

 

bRoot

            [in]a boolean value

 

Remarks

 

            Calling this method adds the currect SSL/TLS session certificate into current user system store.

 

            If bRoot is false, the certificate is added into personal certificate store; If bRoot is true, the certificate is added into trusted root certificate store.
